    def atualizarVida(self, dano):
        if dano > 0 and self.vidaAtual + dano > self.vidaMaxima:
            self.vidaAtual = self.vidaMaxima
        elif dano < 0 and self.vidaAtual + dano <= 0:
            self.vidaAtual = 0
        else:
            self.vidaAtual += dano
    def estaDefendendo(self):
        return self.defendendo

    def defender(self):
        if not self.estaDefendendo():
            self.defesa *= 2
            self.defendendo = True

    def pararDeDefender(self):
        if self.estaDefendendo():
            self.defesa = int(self.defesa * 0.5)
            self.defendendo = False
